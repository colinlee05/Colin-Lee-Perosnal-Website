Password Strength Analyzer

Built a Java tool to assess password security, now live here. Test it out!

document.getElementById('check-btn').addEventListener('click', function() {
    var password = document.getElementById('password-input').value;
    var bar = document.getElementById('strength-bar');
    var strength = 0;

    if (password.length === 0) {
        strength = 0;
    } else if (password.length < 6) {
        strength = 25;
    } else if (password.length < 10 && !/[!@#$%^&*]/.test(password)) {
        strength = 50;
    } else if (password.length >= 10 && /[!@#$%^&*]/.test(password)) {
        strength = 100;
    } else {
        strength = 75;
    }

    bar.style.width = strength + '%';
    bar.style.backgroundColor = strength <= 25 ? '#ff4d4d' : strength <= 50 ? '#ffcc00' : strength <= 75 ? '#66cc00' : '#00cc00';
});
